## 快速输入输出

```cpp
struct In {
    static const int N = 150e6;
    char s[N], *p;
    size_t c;
    In() { c = fread(s, 1, N, stdin), p = s; }
} in;
template <typename T>
In& operator>>(In &i, T &x) {
    char *&p = i.p;
    while (*p != '-' && (*p < '0' || *p > '9')) p++;
    if (p == i.s + i.c) return i;
    bool sgn = false;
    if (*p == '-') sgn = true, p++;
    for (x = 0; *p >= '0' && *p <= '9'; p++) x = x * 10 + *p - '0';
    if (sgn) x = -x;
    return i;
}
In& operator>>(In &i, char *x) {
    char *&p = i.p;
    while (*p == ' ' || *p == '\n') p++;
    if (p == in.s + in.c) return i;
    char *t = p;
    while (*p != ' ' && *p != '\n') p++;
    copy(t, p, x);
    return i;
}
In& operator>>(In &i, string &x) {
    char *&p = i.p;
    while (*p == ' ' || *p == '\n') p++;
    if (p == in.s + in.c) return i;
    char *t = p;
    while (*p != ' ' && *p != '\n') p++;
    x.assign(t, p);
    return i;
}
In& operator>>(In &i, char &x) {
    char *&p = i.p;
    while (*p == ' ' || *p == '\n') p++;
    if (p == i.s + i.c) return i;
    x = *p++;
    return i;
}
struct Out {
    char s[int(15e6)];
    int c = 0, p;
    ~Out() { fwrite(s, 1, c, stdout); }
} out;
Out& operator<<(Out &o, char x) {
    o.s[o.c++] = x;
    return o;
};
Out& operator<<(Out &o, int x) {
    if (x < 0) x = -x, o.s[o.c++] = '-';
    for (o.p = o.c; o.s[o.c++] = x % 10 + '0', x /= 10;);
    reverse(o.s + o.p, o.s + o.c);
    return o;
}
```

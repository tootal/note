    int t;

    while (t--) {
        int n, m;
        cin >> n >> m;

## [B. Prime Square](https://codeforces.com/contest/1436/problem/B)

### 代码
[提交链接](https://codeforces.com/contest/1436/submission/96550321)

```cpp
#include <bits/stdc++.h>
using namespace std;
const int N = 10005;
int np[N];
void pre() {
    np[0] = np[1] = 1;
    for (int i = 2; i < N; i++) {
        for (int j = 2; i * j < N; j++) np[i * j] = 1;
    }
}
int main() {
    pre();
    int t;
    cin >> t;
    while (t--) {
        int n;
        cin >> n;
        for (int i = 2; i < N; i++) {
            if (!np[i] && ((i-1) % (n-1) == 0) && np[(i-1) / (n-1)]) {
                int k = (i-1) / (n-1);
                for (int u = 0; u < n; u++) {
                    for (int v = 0; v < n; v++) {
                        if (u == v) cout << 1;
                        else cout << k;
                        cout << ' ';
                    }
                    cout << '\n';
                }
                break;
            }
        }
    }
}
```

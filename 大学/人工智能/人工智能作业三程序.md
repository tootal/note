# 人工智能作业三程序

```cpp
#include <bits/stdc++.h>
using namespace std;
const int n = 5, m = 5;
const int C[n+1][m+1] = {
    {0, 0, 0, 0, 0, 0},
    {0, 5, 7, 11, 18, 19},
    {0, 18, 22, 15, 15, 6},
    {0, 19, 3, 10, 5, 3},
    {0, 0, 9, 37, 18, 12},
    {0, 1, 12, 5, 6, 14}
};
int dm;
int xs = 1, ys = 2, xd = 5, yd = 5;
stack<pair<int, int>> st;
void print() {
    auto st1 = st;
    vector<pair<int, int>> ans;
    while (!st1.empty()) {
        ans.push_back(st1.top());
        st1.pop();
    }
    reverse(ans.begin(), ans.end());
    for (auto [x, y] : ans) {
        printf("(%d, %d) --> ", x, y);
    }
    cout << '\n';
}
void dfs(int x, int y, int d) {
    if (x < 1 || x > n || y < 1 || y > n) return ;
    d += C[x][y];
    if (d >= dm) return ;
    // printf("(%d, %d)\n", x, y, d);
    st.emplace(x, y);
    if (x == xd && y == yd) {
        dm = d;
        printf("%d: ", d);
        print();
    }
    dfs(x+1, y, d);
    dfs(x-1, y, d);
    dfs(x, y+1, d);
    dfs(x, y-1, d);
    st.pop();
}
int main() {
    for (int i = 1; i <= n; i++)
        for (int j = 1; j <= m; j++)
            dm += C[i][j];
    dfs(xs, ys, 0);
    cout << dm << '\n';
    return 0;
}
```

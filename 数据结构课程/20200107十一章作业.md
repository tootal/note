## 11.15
```cpp
void DFS(Graph *G , int v){
	PreVisit(G,v);
	G->setMark(v,visited);
	for(int w=G->first(v);w<G->n();w=G->next(v,w))
		if(G->getMark(w)==unvisited)
			DFS(G,w);
		else if(G->getMark(w)==visited)
			cout<<”有回路”<<endl;
			return;
	PostVisit(G,v);
}
```
